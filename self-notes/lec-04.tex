\subsection{Sample path properties of Brownian Motion}

\begin{defn}
    \textbf{Holder continuous}\nl
    $f : [0, T] \to \R$ is uniformly Holder continuous with exponent $0 < \gamma \le 1$ if $|f(s) - f(t)| \le k |s - t|^\gamma$ for all $s, t \in [0, T]$.
\end{defn}

\begin{defn}
    \textbf{Holder continuous at some point}.\nl
    $f$ is Holder continuous at point $s_0 \in [0, T]$ if $\exists k$ such that $|f(t) - f(s_0)| \le k |t-s_0|^\gamma$ for all $t \in [0, T]$.
\end{defn}
